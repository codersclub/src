ак поместить приложение в автозапуск Windows

ƒл€ этого надо добавить ключ в реестр:

procedure SetAutorun(aProgTitle,aCmdLine: string;
                     aRunOnce: boolean );
var
  hKey: string;
  hReg: TRegIniFile;
begin
  if aRunOnce then hKey := 'Once'
  else
    hKey := '';

  hReg := TRegIniFile.Create( '' );
  hReg.RootKey := HKEY_LOCAL_MACHINE;
  hReg.WriteString('Software\Microsoft\Windows\CurrentVersion\Run'
                  + hKey + #0,
                  aProgTitle,
                  aCmdLine );
  hReg.destroy;
end;
