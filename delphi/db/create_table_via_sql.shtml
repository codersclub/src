Как создать таблицу через SQL

Следующая функция полностью совместима с Paradox, а так же другими базами данных:

procedure TForm1.Button1Click(Sender: TObject);
var
   Q: TQuery;
begin
     Q := TQuery.Create(Application)
     try
        Q.DatabaseName := 'SF';
        
        with Q.SQL do begin
             Add('Create Table Funcionarios');
             Add('( ID      AutoInc,');
             Add('  Name    Char(30),');
             Add('  Salary  Money,');
             Add('  Depno    SmallInt,');
             Add('  Primary Key ( ID ) )');
        end;
    
        Q.ExecSQL;
     finally
        Q.Free;
     end;
end;
