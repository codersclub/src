Как открыть новое окно, а потом написать в нём что-нибудь

<html>
<head>
<script language = javascript>
function openwindow(){
  w=open('','windowname','width=600,height=250,scrollbars,resizable,toolbar,status');
  with(w.document){
    write("<body>");
    write("Это новое окно");
    write("</body>");
  }
  return false;
 }
}
</script>
</head>
<body>
<input type=button value="Click me to open a new window" onclick="openwindow()">
</body>
</html>
