Таким скриптом можно попугать посетителя вашей странички. Если человек проведёт мышкой по картинке:
то может ненароком подумать, что у него сотрясение мозга :)

  Листинг (можно вставлять в любое место вашей страницы) 
<SCRIPT language=JavaScript1.2><!--
function boom(n)
 {if (window.top.moveBy)
 {for (i = 10; i > 0; i--)
 {for (j = n; j > 0; j--)
 {window.top.moveBy(0,i);
  window.top.moveBy(i,0);
  window.top.moveBy(0,-i);
  window.top.moveBy(-i,0);}}}}
function callJS(jsStr)
{return eval(jsStr)} //--></SCRIPT>
<A name=boom onmouseover="callJS('boom(1)');">
