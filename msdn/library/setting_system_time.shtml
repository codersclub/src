Устанавливаем системное время.

Следующий пример устанавливает системное время при помощи функции SetSystemTime.

 

// Возвращаемое значение - в случае успеха TRUE, иначе FALSE.
// hour     - часы (0-23)
// minutes  - минуты (0-59)

BOOL SetNewTime(WORD hour, WORD minutes)
{
    SYSTEMTIME st;
    char *pc;

    GetSystemTime(&st);       // получаем текущее время
    st.wHour = hour;          // устанавливаем часы
    st.wMinute = minutes;     // и минуты
    if (!SetSystemTime(&st))  // устанавливаем системное время
        return FALSE;
    return TRUE;
}
