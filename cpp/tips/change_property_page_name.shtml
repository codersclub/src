Как изменить имя Property Page

CMyPropertySheet::ChangePageTitle()
{
   CTabCtrl* pTab = GetTabControl();
   TCITEM TabCtrlItem;
   memset( &TabCtrlItem, 0, sizeof(TabCtrlItem));
   TabCtrlItem.pszText = _T("KuKU");
   TabCtrlItem.cchTextMax = _tcslen(TabCtrlItem.pszText);
   TabCtrlItem.mask = TCIF_TEXT;
   pTab->SetItem( 5, &TabCtrlItem );
}
