Пример использования CFile

Следующий пример открывает и читает файл при помощи MFC класса CFile. См Работа с файлами в документации MFC по использованию CFile.

    char Buffer[100];
    CFile File;
    CFileException fe;
if (!File.Open(pszPathName, CFile::modeRead, &fe)) {
    fe->ReportError();
    return FALSE;
    }
TRY {
    while (File.Read(&Buffer, sizeof(Buffer))) {
        }
    File.Close();
    }
CATCH_ALL(e) {
    File.Abort(); // will not throw an exception
    e->ReportError();
    return FALSE;
    }
END_CATCH_ALL
