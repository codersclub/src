int CInspectorApp::ExitInstance() 
{
	GXTerminate();		// terminates objective grid (fine product!)

	if (m_pDB)			// Application holds both the primary and temp dbs	
	{
		if(m_pDB->IsOpen())
			m_pDB->Close();
		delete m_pDB;
	}

	if (m_pTempDB)
	{
		if (m_pTempDB->IsOpen())
			m_pTempDB->Close();
		delete m_pTempDB;
	}
	
	return CWinApp::ExitInstance();
}
