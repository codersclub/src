Try:

CDockBar* pDock = pYourToolBar->m_pDockBar;
if( pDock )
{
    switch( pDock->GetDlgCtrlId())
    {
        case AFX_IDW_DOCKBAR_LEFT:
        /*...*/

        case AFX_IDW_DOCKBAR_RIGHT:
        /* and so on: TOP & BOTTOM. */

        default:
            ASSERT( FALSE );
    }
}
else
    TRACE("CMyToolBar is not docked\n");
