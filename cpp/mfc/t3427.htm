Q:  How can I get the HWND to the CHeaderCtrl that the CListCtrl is using?

A:  CListCtrl* pListCtrl = (CListCtrl *) GetDlgItem(IDC_YOURLISTVIEW);
      ASSERT(pListCtrl != NULL);
   
      CHeaderCtrl* pHeaderCtrl = (CHeaderCtrl *) pListCtrl->GetDlgItem(0);
      ASSERT(pHeaderCtrl != NULL);
