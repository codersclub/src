Here is the code to get all the docs and views from the cwinapp class. 

Void YourApp::OnGetAllDocsAndViews() {
 POSITION pos = GetFirstDocTemplatePosition();
 while (pos!= NULL) {  // Iterate through all doc templates
  CDocTemplate* pDocTemplate = GetNextDocTemplate(pos);
  POSITON docPos = pDocTemplate->GetFirstDocPosition() ;
  while (docPos != NULL) {  // all Documents in a doc template
   pDoc = pDocTemplate->GetNextDoc(docPos);
   POSITION viewPos = pDoc->GetFirstViewPostion();
   while ( viewPos != NULL) {// All the views attached to the document
    CView* pView = pDoc->GetNextView(viewPos);
    //  here you can do your clean up
    //
   }
  }   
 }
}
