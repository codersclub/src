But you can make it a dialog, which you embed on your pages.
Something like this:

class CMyControls : public CDialog
{
  ...
}

class CPage1 : public CPropertyPage
{
    ...
    CMyControls m_MyControls;
    ...
}

CPage1::OnInitialUpdate()
{
   VERIFY(m_MyControls.Create(IDD_MYCONTROLS,this));

   CRect Rect;
   GetDlgItem(IDC_PLACEHOLDER_FRAME)->GetWindowRect(Rect); // invisible
place holder
   ScreenToClient(Rect);

   #ifdef _DEBUG
        // Check that it fits in the place holder
        CRect controlrect;
        VERIFY(m_MyControls.GetClientRect(controlrect));
        ASSERT(controlrect.Width() <=3D Rect.Width() && controlrect.Heigh=

   #endif

|SWP_NOACTIVATE));
