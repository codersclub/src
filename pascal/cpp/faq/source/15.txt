§NAME: Выключение компьютьера
§AUTHOR: server_mouse
§EMAIL: server_mouse@mail.ru

Используется API-функция 
[code]bool ExitWindowsEx (UINT uFlags, DWORD dwReason);[/code]
[gray]Declared in Winuser.h; include Windows.h.
Use User32.lib.[/gray]

[b][u]uFlags:[/u][/b]
[i]для перегрузки используем [b]EWX_REBOOT[/b]
для выключения питания [b]EWX_POWEROFF[/b]
если нужно экстренно выключит/перегрузить комп, используем вышеуказаные влаги в сочетании с [b]EWX_FORCE[/b][/i]

Второй параметр используется только в XP для указания причины перезагрузки.

В NT/2000/XP для использования [i]ExitWindowsEx[/i] необходимо иметь
привелегию SE_SHUTDOWN_NAME, по умолчанию не установленную
даже для администратора. Установить её можно нижеприведённым кодом. 
[code]HANDLE hToken;
TOKEN_PRIVILEGES* NewState;
OpenProcessToken (GetCurrentProcess (), TOKEN_ADJUST_PRIVILEGES, 
                 &hToken);
NewState = (TOKEN_PRIVILEGES*) malloc (sizeof 
             (TOKEN_PRIVILEGES) + sizeof (LUID_AND_ATTRIBUTES));
NewState->PrivilegeCount = 1;
LookupPrivilegeValue (NULL, SE_SHUTDOWN_NAME, 
                         &NewState->Privileges[0].Luid);
NewState->Privileges[0].Attributes = SE_PRIVILEGE_ENABLED;
AdjustTokenPrivileges (hToken, FALSE, NewState, NULL, NULL, NULL);
free (NewState);
CloseHandle (hToken);[/code]
