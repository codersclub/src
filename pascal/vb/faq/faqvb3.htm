Q: Как сменить рисунок рабочего стола?
A (=Alex=):

Private Declare Function SystemParametersInfo Lib "user32" Alias "SystemParametersInfoA" (ByVal uAction As Long, _
ByVal uParam As Long, ByVal lpvParam As String, ByVal fuWinIni As Long) As Long

Const SPI_SETDESKWALLPAPER = 20

Private Sub Command1_Click()
SystemParametersInfo SPI_SETDESKWALLPAPER, 0, "D:\Basic\tmpProj\Load.bmp", True
'Заменить путь D:\Basic\tmpProj\Load.bmp на нужный вам файл рисунка в формате bmp
End Sub

Private Sub Command2_Click()
SystemParametersInfo SPI_SETDESKWALLPAPER, 0, 0, False
End Sub
