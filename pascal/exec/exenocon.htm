Procedure ExecNoCon(Prog,Param,ToOut : String);
Var NDir    : Array[1..80] of Byte;
    SSt     : String;
    I       : Integer;
    F       : Text;
    Reg     : Registers;
    Desc,
    OldDisc : Word;
Begin
 Sst:=ToOut;
 Sst:=Sst+#0;
 For I:=1 To Length(Sst) Do
  Ndir[I]:=Ord(Sst[I]);
 Assign(F,ToOut);    { СЃРёРµ РґР»СЏ СЃРѕР·РґР°РЅРёСЏ С„Р°Р№Р»Р° РЅСѓР»РµРІРѕР№ РґР»РёРЅС‹ }

 Reg.AH:=$3D;        { РѕС‚РєСЂС‹С‚РёРµ С„Р°Р№Р»Р° СЃСЂРµРґСЃС‚РІР°РјРё DOS, РґР°Р»СЊС€Рµ РІСЃРµ РїСЂРѕР·СЂР°С‡РЅРѕ }
 Reg.AL:=2;
 Reg.DS:=Seg(Ndir);
 Reg.DX:=Ofs(Ndir);
 MsDos(Reg);
 Desc:=Reg.AX;
 {Duplicate Discriptor}
 Reg.AH:=$45;
 Reg.BX:=1;
 MsDos(Reg);
 Olddisc:=Reg.AX;
 {ReDirect Discriptor}
 Reg.AH:=$46;
 Reg.BX:=Desc;
 Reg.CX:=1;
 MsDos(Reg);
 {Close File}
 Reg.AH:=$3E;
 Reg.BX:=Desc;

 {Execute File}

 Reg.AH:=$46;
 Reg.BX:=Olddisc;
 Reg.CX:=1;

 {Close Olddisc}
 Reg.AH:=$3E;
 Reg.BX:=Olddisc;
 MsDos(Reg);
End;
