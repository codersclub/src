'to van:
'вот примерчик бегущей строки:
'на форму кнопку (command1), текстовое поле (text1), надпись(label1) и таймер(timer1)
'КОД:
Private Type snaplab
label As String
End Type
Private mega As snaplab
Public Sub go1(lab)
label1.Caption = ""
mega.label = lab
timer1.Enabled = -1
End Sub
Private Sub Form_Unload(Cancel As Integer)
End
End Sub
Private Sub timer1_Timer()
a1 = Len(mega.label)
a2 = Len(label1.Caption)
If a1 = a2 Then
timer1.Enabled = False
Exit Sub
End If
a3 = Left(mega.label, a2 + 1)
label1.Caption = a3
End Sub
Private Sub command1_Click()
go1 text1.Text
End Sub
Private Sub Form_load()
text1.Width = 1.222
text1.Height = 285
text1.Top = 120
text1.Left = 0.049
command1.Width = 0.638
command1.Height = 375
command1.Left = 1.313
command1.Top = 120
form1.Width = 4960
form1.Height = 2115
form1.Top = 1500
form1.Left = 1500
label1.AutoSize = 0
label1.Height = 1200
label1.Width = 1.915
label1.Left = 0.049
timer1.Enabled = 0
timer1.Interval = 100
Me.Caption = "Пример бугущей строки"
command1.Caption = "GO"
text1.Text = "Пример бугущей строки... Можешь поменять шрифт у label1, как тебе хочется!.." & vbCrLf & " Исходя из этого примера можно сделать действительно бегущую строку:)"
End Sub
