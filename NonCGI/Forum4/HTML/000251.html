cr опубликован 11-01-2002 09:48 MSK            
Private Sub Form_Load()
Text1.Text = ""
End Sub

Private Sub Text1_KeyPress(KeyAscii As Integer)
Dim aTemplate As String
aTemplate = "a"
dTemplate = "d"
If InStr(1, aTemplate, Chr(KeyAscii)) > 0 Then KeyAscii = 63
If InStr(1, dTemplate, Chr(KeyAscii)) > 0 Then KeyAscii = 37
'и т.д. и т.п.
End Sub
