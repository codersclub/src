Скриншот можно сделать так:

1. Cоздать на форме PictureBox (можно сделать его невидимым)

далее программно:

2. Эмулировать нажатие клавиш Print Screen (весь экран) или Alt+Print Screen (одно активное окно)

Private Declare Sub keybd_event Lib "user32" (ByVal bVk As Byte, ByVal bScan As Byte, _
ByVal dwFlags As Long, ByVal dwExtraInfo As Long)

Const VK_MENU = &H12
Const KEYEVENTF_KEYUP = &H2
Const VK_SNAPSHOT = &H2C

Private Sub ShowStartMenu()
Call keybd_event(VK_MENU, 0, 0, 0) 'Hажимаем Alt
Call keybd_event(VK_SNAPSHOT, 0, 0, 0) 'Hажимаем Print Screen
Call keybd_event(VK_SNAPSHOT, 0, KEYEVENTF_KEYUP, 0) 'Отпускаем Print Screen
Call keybd_event(VK_MENU, 0, KEYEVENTF_KEYUP, 0) 'Отпускаем Alt
' End
End Sub

Private Sub Command1_Click()
ShowStartMenu
End Sub

3. Вставить в пикчербокс картинку из буфера
Picture1.Picture = Clipboard.GetData

4. Сохранить картинку из пикчербокс'а
SavePicture Picture1.Picture, "C:\temp\CoolPicture.bmp"
