Вроде, примерно так должно получиться:

if (document.images) {
name1on = new Image();
name1on.src = 'name1on.gif';
name1off = new Image();
name1off.src = 'name1off.gif';

name2on = new Image();
name2on.src = 'name2on.gif';
name2off = new Image();
name2off.src = 'name2off.gif';
................
// можно при желании сделать и массив:)

}
function Over(name) {
if (document.images)
document.images[name].src = eval(name + 'on.src');
}

function Out(name) {
if (document.images) document.images[name].src = eval(name + 'off.src');
}


p.s. Но еще лучше использовать какой-нибудь Java-апплетик соответствующий;)
