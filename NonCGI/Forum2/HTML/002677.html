Попробуй что-то вроде:

bool MakeShortcut(LPCSTR title, LPCSTR filename, LPCSTR workdir) {

bool st = FALSE;

char keyval[MAX_PATH];
DWORD keyvallen = sizeof(keyval);
unsigned short fn[MAX_PATH];

HKEY hKey;

if (RegOpenKeyEx(HKEY_CURRENT_USER, "Software\\Microsoft\\Windows\\CurrentVersion\\Explorer\\Shell Folders",
0, KEY_QUERY_VALUE, &hKey) != ERROR_SUCCESS)
return st;

if (RegQueryValueEx(hKey, "Desktop", 0, 0, (unsigned char *)keyval, &keyvallen) == ERROR_SUCCESS)
st = TRUE;

RegCloseKey(hKey);

if (!st)
return st;

st = FALSE;

lstrcat(keyval, "\\");
lstrcat(keyval, title);
lstrcat(keyval, ".lnk");

MultiByteToWideChar(CP_ACP, 0, keyval, -1, fn, MAX_PATH);

CoInitialize(NULL);

IShellLink * psl;
IPersistFile * pIPF;
HRESULT hr = ::CoCreateInstance(CLSID_ShellLink, NULL, CLSCTX_INPROC_SERVER, IID_IShellLink, (void **)&psl);

if (SUCCEEDED( hr )) {
psl->SetPath(filename);
psl->SetDescription(title);
psl->SetIconLocation(filename, 0);
psl->SetWorkingDirectory(workdir);

hr = psl->QueryInterface(IID_IPersistFile, (void**)&pIPF);
if (SUCCEEDED(hr)) {
hr = pIPF->Save(fn, TRUE);
if (hr == S_OK)
st = TRUE;
pIPF->Release();
}
psl->Release();
}

CoUninitialize();

return st;
}
