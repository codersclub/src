Автор Тема:   Вложенные Dll-ки
Drunkard опубликован 03-02-2002 22:41 MSK        
Программа Пускач загружает Dll в которой организуется поток запускающий таймер.
Если Пускач загружает одну Dll, то все ОК. Но необходимо из запущеной DLL запустить еще одну,
аналогичную первой, получаются вложенные вызовы. Коды всех трех программ приведены ниже,
это Пускач, загружающий Dll_1, Dll_1, загружающая Dll_2 и сама Dll_2. Код рабочий. Таймеры срабатывают, все ОК.
Загвоздка в выгрузке. Создаются такие ситуёвины:
1) При нажатии EXIT(CM_EXIT), DLL_1 выдает сообщение в обработчике ошибок (см. место в коде DLL_1):
"Не могу выгрузить библиотеку DLL_2. Ошибка 0: Операция выполнена успешно." :))
После этого, обе DLL-ки выдают сообщения "Закрываюсь" и Пускач тоже завершается нормально.
2) При нажатии "Выгрузить"(CM_DLLPURGE), DLL_1 выдает сообщение в обработчике ошибок (см. место в коде DLL_1):
"Не могу выгрузить библиотеку DLL_2. Ошибка 87: Параметр задан неверно."
После этого, обе DLL-ки выдают сообщения "Закрываюсь", а Пускач продолжает нормально фунциклировать.
То есть, если не выводить на экран сообщения об ошибках, то получится всё ОК. Прога вроде бы рабочая, но волнуют эти два момента.
Что за хрень, могет кто знает?
Если из Dll_1 убрать FreeLibrary(DLL_2), то по команде "Выгрузить", DLL_1 выгрузится, а DLL_2 будет работать до завершения Пускача и
ничем ее не вытравишь.
Код довольно простой и каждый может убедиться в этих выкрутасах.
Кто знает в чем дело?

PS Коды Borland-овские

Кусок кода из Пускача:
(В нем три кнопочки "Загрузить" - CM_DLLLOAD и "Выгрузить" - CM_DLLPURGE и "EXIT")
/////////////////////////////////////////////////////
case CM_DLLLOAD:
hDll_1=LoadLibrary("DLL_1");
break;
case CM_DLLPURGE: //Завершаем dll_1
if(hDll_1) FreeLibrary(hDll_1);
break;
case CM_EXIT:
DestroyWindow(hwnd);
break;
////////////////////////////////////////////////////
Dll_1 : // Вызывающая прога

HINSTANCE hDll_2=NULL;
UINT TimerId=0;

BOOL Maino(void);
void React_OnTimer(void);

BOOL WINAPI DllEntryPoint (HINSTANCE hInst,DWORD Reason, LPVOID)
{
DWORD ThId;
HANDLE hThread;
BOOL AttOK;
char Str[MAX_PATH+80],BufErr[80];
DWORD Error;

switch(Reason) {
case DLL_PROCESS_ATTACH:// Запуск потока
hThread=CreateThread(NULL,0,(LPTHREAD_START_ROUTINE)Maino,NULL,0,&ThId);
if(!hThread) AttOK=FALSE;
else {
AttOK=TRUE;
hDll_2=LoadLibrary("Dll_2");
}
break;
case DLL_PROCESS_DETACH:
if(hDll_2) {
if(!FreeLibrary(hDll_2)) {// В этом месте выдает ошибки. См. описание
Error=GetLastError();
FormatMessage(FORMAT_MESSAGE_FROM_SYSTEM,NULL,Error,LANG_SYSTEM_DEFAULT,BufErr,80,NULL);
sprintf(Str,"Не могу выгрузить библиотеку DLL_2\nОшибка %u: %s",Error,BufErr);
MessageBox(0,Str,"DLL_1",MB_OK);
}
else hDll_2=NULL;
};
if(TimerId) KillTimer(NULL,TimerId);
MessageBox(0,"Закрываюсь","Dll_1",MB_OK);
break;
}
return AttOK;
}

BOOL Maino(void)
{
MSG msg;

TimerId=SetTimer(NULL,1,(UINT)10000,NULL);// Таймер на 10 сек
while(GetMessage(&msg,NULL,0,0)) {
if (msg.message==WM_TIMER) { //Ловим сообщения от таймера
if(msg.wParam==(WPARAM)TimerId) React_OnTimer();
};
TranslateMessage(&msg);
DispatchMessage(&msg);
}
return TRUE;
}

void React_OnTimer(void)
{
MessageBox(0,"Сработал таймер","Dll_1",MB_OK);
}
//////////////////////////////////////////////////
Dll_2: // Вызываемая

HANDLE hEvent;
UINT TimerId=0;

BOOL Maino(void);
void React_OnTimer(void);

BOOL WINAPI DllEntryPoint (HINSTANCE hInst,DWORD Reason, LPVOID)
{
DWORD ThId;
HANDLE hThread;
BOOL AttOK;

switch(Reason) {
case DLL_PROCESS_ATTACH:
hThread=CreateThread(NULL,0,(LPTHREAD_START_ROUTINE)Maino,NULL,0,&ThId);
if(hThread) AttOK=TRUE;
else AttOK=FALSE;
break;
case DLL_PROCESS_DETACH:
if(TimerId) KillTimer(NULL,TimerId);
MessageBox(0,"Закрываюсь","Dll_2",MB_OK);
break;
}
return AttOK;
}

BOOL Maino(void)
{
MSG msg;

TimerId=SetTimer(NULL,1,(UINT)15000,NULL);// Таймер на 15 сек
while(GetMessage(&msg,NULL,0,0)) {
if (msg.message==WM_TIMER) { //Ловим сообщения от таймера
if(msg.wParam==(WPARAM)TimerId) React_OnTimer();
};
TranslateMessage(&msg);
DispatchMessage(&msg);
}
return TRUE;
}

void React_OnTimer(void)
{
MessageBox(0,"Сработал таймер","Dll_2",MB_SYSTEMMODAL|MB_OK);
}

Во первых помимо PROCESSATTACH/PROCESSDETACH обрабатывай THREADATTACH/THREADDETACH. Пролема может быть здесь. То есть, перед выгрузкой DLL все нити, которые ее используют, должны быть остановлены.
Воторое - вынеси нафиг весь код, отвечающий за запуск нитей и т. п., из функций DLL_PROCESS_ATTACH/DETACH. Это плохой стиль. Лучше после загрузки DLL вызывай соответвующие функции. Ведь ты не можешь заранее предположить, кем будет грузиться твой DLL - может быть это будет сканер addon'ов, загружающий все DLL'ки подряд.
