#include <stdlib.h>
#include <stdio.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <sys/ioctl.h>

int main(int argc, char* argv[])
{
int listener;
sockaddr_in sin;
in_addr addr;
unsigned long argp;
int s1, connected;
sockaddr addr1;
listener = socket(AF_INET,SOCK_STREAM,IPPROTO_TCP);
addr.s_addr = 0;
memset(&sin, 0, sizeof(sin));
sin.sin_family = AF_INET;
sin.sin_addr = addr;
sin.sin_port = htons(5051);
ioctl(listener,FIONBIO,&argp);
bind(listener,(struct sockaddr*) &sin,sizeof(sin));
listen(listener,10);
socklen_t addr_len = sizeof(addr1);

char *recieved = (char *)malloc(2);

while (true)
{
s1=accept(listener,&addr1,&addr_len);
if (s1 != -1)
{
connected = s1;
};
if (recv(connected,recieved,1,0)==1)
{
recieved[1] = NULL;
printf(recieved);
};
};

}
