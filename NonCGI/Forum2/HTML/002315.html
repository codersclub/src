{
char er[]="Неправильно!!!\n";

bool error=false;
bool tochka=false;
bool valueX=false;
int lScobka=0;int lScobkaPos[15];
int rScobka=0;int rScobkaPos[15];
int lModule=0;int lModulePos[15];
int rModule=0;int rModulePos[15];
int lSQR=0;int lSQRPos[15];
int rSQR=0;int rSQRPos[15];
while(string[j]!='\0'){
switch(string[j]){
case 'x':
valueX=true;
break;
case '=':
if(ravno){error=true;}
ravno=true;
break;
case ';':
if(tochka){error=true;}

break;

if(lSQR>15)error=true;
break;
case '}':
rSQR++;
rSQRPos[rSQR]=j;
if(rSQR>15)error=true;
break;

}
j++;
