BOOL GetProcAddresses(HINSTANCE *hLibrary, LPCSTR lpszLibrary, INT nCount,...)
{
va_list va;
va_start( va, nCount );

if ( ( *hLibrary = LoadLibrary( lpszLibrary ) ) != NULL )
{
FARPROC * lpfProcFunction = NULL;
LPSTR lpszFuncName = NULL;
INT nIdxCount = 0;
while ( nIdxCount < nCount )
{
lpfProcFunction = va_arg( va, FARPROC* );
lpszFuncName = va_arg( va, LPSTR );
if ( ( *lpfProcFunction = GetProcAddress( *hLibrary,
lpszFuncName ) ) == NULL )
{
lpfProcFunction = NULL;
return FALSE;
}
nIdxCount++;
}
}
else
{
va_end( va );
return FALSE;
}
va_end( va );
return TRUE;
}
,

Если я не ошибаюсь FARPROC - это указатель на функцию, WNETCLOSEENUM - тоже указатель на функцию. А теперь посмотри внимательно, что ты передаешь в функцию GetProcAddresses, которая принимает указатель на указатель на функцию. Естественно, она у тебя вылетает по Access Violation.

PS: Функции с переменным числом аргументов надо использовать с ОЧЕНЬ БОЛЬШОЙ ОСТОРОЖНОСТЬЮ!!!!! По скольку не происходит абсолютно никакой проверки на корректность типов переданых параметров.
