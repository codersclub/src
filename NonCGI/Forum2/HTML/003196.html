Есть сабж, в котором скрипт вызывает метод(он возвращает тип VARIANT). После некоторых заморочек в скрипте, вызывается метод с аргументом LPCTSTR (опять же из скрипта). Ему передаются данные, но они - ИЗУРОДОВАНЫ!!!

Вот пример скрипта:


function Publish()
{
var FilesRecordset = OpenRecordset("select * from files where NOT file=\'\'");
if (FilesRecordset >= 0)
{
while (!IsEOF(FilesRecordset))
{
var fileName = GetDataField(FilesRecordset, "file");
var fileID = GetDataField(FilesRecordset, "id");
CopyFile(fileName, fileID, "");
var objectWidth = GetObjectWidth(fileName);
var objectHeight = GetObjectHeight(fileName);
var outputFileName = fileID + ".html";
var outputFile = CreateFile(outputFileName);
var fileSign = GetDataField(FilesRecordset, "sign");
if (fileSign != "")
{
var temp = fileName + " (" + fileSign + ") ";
MessageBox(temp); // показывает полное уродство!!!
fileName = temp;
}

// заголовок html-файла
Write(outputFile, ("<HTML><HEAD><TITLE>" + GetDataField(FilesRecordset, "name") + "</TITLE></HEAD><BODY>"));
// рисунок
MessageBox(fileID);
Write(outputFile, "<IMG SRC=\"images/" + fileID + GetFileExtension(fileName) + "\">");
// окончание html-файла
Write(outputFile, "<BR></BODY></HTML>");
CloseFile(outputFile);
MoveNext(FilesRecordset);
}
CloseRecordset(FilesRecordset);
}
return true;
}
