Я использую очччень простой способ. Прямо первыми строками проги пишешь:

// Устанавливаем событие, что программа загрузилась
if(CreateEvent(NULL,FALSE,TRUE,"Моя прога")) {
// Если данное событие уже зарегистрировано, то вываливаемся
if(GetLastError()==ERROR_ALREADY_EXISTS) {
MessageBox(0,"Сначала завершите работу предыдущей проги!","Название",MB_SYSTEMMODAL|MB_OK);
return FALSE;
};
}
else return FALSE;
