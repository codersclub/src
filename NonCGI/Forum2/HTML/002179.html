michl_m опубликован 05-12-2001 14:15 MSK            
purpe прав. С МАС адресом идея неплохая.
Для его получения нужно использовать
функцию Netbios(). Так можно получить
Ethernet Number on LANA.
Примерно так:

struct __ADAPTER_STATUS
{
ADAPTER_STATUS adapt;
NAME_BUFFER NameBuff [30];
} Adapter;

BOOL GetEtherNumber(char* NetName)
{
NCB Ncb;
LANA_ENUM lenum;

memset( &Ncb, 0, sizeof(Ncb) );
Ncb.ncb_command = NCBENUM;
Ncb.ncb_buffer = (UCHAR *)&lenum;
Ncb.ncb_length = sizeof(lenum);
Netbios( &Ncb );

if ( lenum.length == 0 ) return FALSE;


memset( &Ncb, 0, sizeof(Ncb) );
Ncb.ncb_command = NCBRESET;
Ncb.ncb_lana_num = lenum.lana[0];
Netbios( &Ncb );

memset( &Ncb, 0, sizeof (Ncb) );
Ncb.ncb_command = NCBASTAT;
Ncb.ncb_lana_num = lenum.lana[0];

memset(Ncb.ncb_callname,' ', NCBNAMSZ);
Ncb.ncb_callname[0]='*';

Ncb.ncb_buffer = (unsigned char *)&Adapter;
Ncb.ncb_length = sizeof(Adapter);

UCHAR uRetCode = Netbios( &Ncb );

if ( uRetCode != 0 ) return FALSE;

*NetName = 0;
for(int i=0; i<6; i++)
sprintf(NetName, "%s%02x", NetName, Adapter.adapt.adapter_address[i]);

return TRUE;
}

Затем вызвать:
char NetNameString[128] = "";
BOOL bResult = GetEtherNumber(NetNameString);

Не забудь подлинковать netapi32.lib
и включить #include "nb30.h"
