#include <pthread.h>
#include <stdio.h>


void *print_message_function( void *ptr );

int main(int argc, char **argv)
{
pthread_t thread1, thread2;
char *message1 = "Hello";
char *message2 = "World";
void* retval1,*retval2;

pthread_create( &thread1, 0,
print_message_function, (void *) message1);
//sleep(10);
pthread_create(&thread2, 0,
print_message_function, (void *) message2);
pthread_join(thread1,&retval1);
pthread_join(thread2,&retval2);

printf("ghghg");
//getch();
exit(0);
}

void *print_message_function( void *ptr )
{
char *message;
message = (char *) ptr;
for (int i=0;i<10;i++)
{
printf("(%d) %s %08X\n", i,message,(int)pthread_self());
sched_yield();
}
pthread_exit(0);
return 0;
}
