Вот функция, с помощью которой я беру строки аттрибутов из сертификата. Их может быть две - атрибуты того, кто выдал, и того, кому выдали. Это определяется параметром int SubjIssuer.
Параметр pszObjOid - это OID того, что ты хочешь вытащить. Список OIDов есть в MSDN.
ПРимеры OIDов:
szOID_COMMON_NAME
szOID_ORGANIZATIONAL_UNIT_NAME
szOID_ORGANIZATION_NAME
szOID_LOCALITY_NAME
szOID_COUNTRY_NAME
szOID_RSA_emailAddr

Получив значения этих полей, ты можешь решить -- нужен тебе данный сертификат или нет.
Скажем,
CString sAttr;
sAttr = GetAttrName(cert, szOID_COMMON_NAME, SUBJECT_ATTR);
if(sAttr == "VasjaPupkin")
{
//это он
}

Да, там кое-где встречаются мои макросы - типа _MBNT_(...). Не обращай внимания - это в основном отладка. _MBNT_ это просто MessageBox.

CString GetAttrName(PCCERT_CONTEXT cert,LPCSTR pszObjId, int SubjIssuer)
{
USES_CONVERSION;
CString name;
CERT_NAME_INFO *pDecodeName; // Point variable to hold
// The address of the decoded
// buffer.
DWORD cbDecoded; // Variable to hold the
// length of the decoded buffer.
BYTE *pbDecoded = NULL; // Variable to hold a pointer to the
// decoded buffer.
PCERT_RDN_ATTR attr;
char *aname;
DWORD cbEncoded;
BYTE* pbEncoded;

//Get size
if(SubjIssuer == SUBJECT_ATTR)
{
pbEncoded = cert->pCertInfo->Subject.pbData;
cbEncoded = cert->pCertInfo->Subject.cbData;
}
else if(SubjIssuer == ISSUER_ATTR)
{
pbEncoded = cert->pCertInfo->Issuer.pbData;
cbEncoded = cert->pCertInfo->Issuer.cbData;
}
else
{
return "";
}

if(CryptDecodeObjectEx(
MY_TYPE,
X509_NAME,
pbEncoded, // The buffer to be decoded.
cbEncoded,
CRYPT_DECODE_NOCOPY_FLAG,
NULL,
NULL,
&cbDecoded))
{
//name.Format("Len%d",cbDecoded);
}
else
{
//_MBNT_("Error getting size for decode")
return "";
}
//decode
pbDecoded = new BYTE[cbDecoded];
if(CryptDecodeObjectEx(
MY_TYPE,
X509_NAME,
pbEncoded, // The buffer to be decoded.
cbEncoded,
CRYPT_DECODE_NOCOPY_FLAG,
NULL,
pbDecoded,
&cbDecoded))
{
pDecodeName = (CERT_NAME_INFO*)pbDecoded;
attr = CertFindRDNAttr(
pszObjId,
pDecodeName);
if(attr == NULL)
{
name = "";
}
else
{
if(((BYTE*)attr->Value.pbData)[1] == 0x04) //значит там лежит UNICODE строка. Декодируем ее
{
aname = W2A((const unsigned short*)attr->Value.pbData);
name = aname;
}
else
name = attr->Value.pbData;
// name = Blob2Hex(&attr->Value);
}
}
else
{
//_MBNT_("Error decoding");
return("");
}

_RELEASE_(pbDecoded);
return name;
}

В догонку сервисные функции, которые я использую:
Перебор всех сертификатов в "MY".
_EXPORTFUN int WINAPI GetNextCert(char *StoreName,char *CurrentCertId, int *pNewKid,
int *pSubjectStr, int *pIssuerStr, int *pVersionStr,
int *pFriendlyName, int *pNotBefore, int *pNotAfter)
{
USES_CONVERSION;
//Возвращаемые значения
//0 - все хорошо
//1 - сертификатов больше нет
//2 - ошибка открытия хранилища
//3 - ошибка выделения памяти
//4 - ошибка поиска сертификата. Возвращается если произошла ошибка в GetCert
//255 - неизвестная ошибка
PCCERT_CONTEXT Cert;
HCERTSTORE Store;
FILETIME ftNotBefore;
FILETIME ftNotAfter;
static char buffer[10240];
unsigned long buffersize = 10240;
char* ansibuffer;
CString sCurrentCertId = CurrentCertId;
PCERT_RDN_ATTR attrib = NULL;
PCERT_NAME_INFO infoname = NULL;
static CString sCID;
static CString sSubject;
static CString sIssuer;
static CString sVersion;
static CString sFriendlyName;
static CString sNotBefore;
static CString sNotAfter;
static CString sCommonName;
ZeroMemory(buffer,10240);
try
{
Store = CertOpenSystemStore(NULL,StoreName);
if(!Store)
{
return 2;
////// ОШИБКА
}//if
if(sCurrentCertId.IsEmpty())
Cert = NULL;
else
{
//Cert = GetCert(sCurrentCertId);
Cert = NULL;

//найдем нужный сертификат
again:

Cert = CertEnumCertificatesInStore(
Store,
Cert);
CertNameToStr(
MY_TYPE,
&Cert->pCertInfo->Subject,
CERT_SIMPLE_NAME_STR | CERT_NAME_STR_NO_PLUS_FLAG,
buffer,
10240);
sSubject = buffer;

if(sCID != Blob2Hex(&Cert->pCertInfo->SerialNumber) +
sSubject
&& Cert)
goto again;
if(!Cert)
return 4;

}//else
Cert = CertEnumCertificatesInStore(
Store,
Cert);
if(!Cert)
{
//MessageBox(NULL,"Невозможно получить следующий сертификат","Cryptdll",MB_OK);
CertFreeCertificateContext(Cert);
CertCloseStore(Store,CERT_CLOSE_STORE_FORCE_FLAG);
return 1;
}//if
}//try
catch(my_error error)
{
switch(error.errcode)
{
case ERROR_OPEN_CERT_STORE: return 2;
case ERROR_FINDING_CERTIFICATE: return 4;
case ERROR_MEMORY_ALLOCATION: return 3;
case ERROR_UNKNOWN_ERROR: return 255;
default: return 255;
}//switch
}//catch
//получили следующий сертификат или ушли, если их больше нет
//------------а теперь разгребем сертификат--------------
//получим его серийный номер
sCID = Blob2Hex(&Cert->pCertInfo->SerialNumber); //дальнейший апдейт смотри ниже в заполнении Subject
//получим срок годности
ftNotBefore = Cert->pCertInfo->NotBefore;
ftNotAfter = Cert->pCertInfo->NotAfter;
sNotBefore = FileTime2CString(ftNotBefore);
sNotAfter = FileTime2CString(ftNotAfter);
//получим Subject
//попробуем взять CommonName
//sCommonName = InvertAttrString(Cert,SUBJECT_ATTR);
CertNameToStr(
MY_TYPE,
&Cert->pCertInfo->Subject,
CERT_SIMPLE_NAME_STR | CERT_NAME_STR_NO_PLUS_FLAG,
buffer,
10240);
sSubject = buffer;
sCID+=sSubject;
//обновим sCID
//получим инвертированное значение сабжект
CertNameToStr(
MY_TYPE,
&Cert->pCertInfo->Subject,
CERT_SIMPLE_NAME_STR | CERT_NAME_STR_NO_PLUS_FLAG | CERT_NAME_STR_REVERSE_FLAG,
buffer,
10240);
sCommonName = buffer;
//получим Issuer
CertNameToStr(
MY_TYPE,
&Cert->pCertInfo->Issuer,
CERT_SIMPLE_NAME_STR | CERT_NAME_STR_NO_PLUS_FLAG | CERT_NAME_STR_REVERSE_FLAG,
buffer,
10240);
sIssuer = buffer;

// sIssuer = InvertAttrString(Cert,ISSUER_ATTR);
//получим версию
switch(Cert->pCertInfo->dwVersion)
{
case CERT_V1: sVersion = "1";break;
case CERT_V2: sVersion = "2";break;
case CERT_V3: sVersion = "3";break;
default: sVersion = "?";
}
//получим файл с понятным именем
if(!CertGetCertificateContextProperty(
Cert,
CERT_FRIENDLY_NAME_PROP_ID,
buffer,
&buffersize))
{
if(GetLastError() == CRYPT_E_NOT_FOUND)
sFriendlyName = "";
else
sFriendlyName = "-";
}
else
{
ansibuffer = W2A((unsigned short*)buffer);
sFriendlyName = ansibuffer;
}
//конец работы
CertFreeCertificateContext(Cert);
CertCloseStore(Store,CERT_CLOSE_STORE_FORCE_FLAG);
*pSubjectStr = reinterpret_cast<int>((LPCTSTR)sCommonName);
// *pIssuerStr = reinterpret_cast<int>((LPCTSTR)sIssuer);
*pIssuerStr = reinterpret_cast<int>((LPCTSTR)sIssuer);
*pVersionStr = reinterpret_cast<int>((LPCTSTR)sVersion);
*pNewKid = reinterpret_cast<int>((LPCTSTR)sCID);
*pFriendlyName = reinterpret_cast<int>((LPCTSTR)sFriendlyName);
*pNotBefore = reinterpret_cast<int>((LPCTSTR)sNotBefore);
*pNotAfter = reinterpret_cast<int>((LPCTSTR)sNotAfter);
return 0;
}//GetNextCert
