Учите матчасть... эта идиома (виртуальный конструктор) базируется на том основании что при конструировании объекта на стеке или динамически компилятор ВСЕГДА вызывает конструктор класса. т.е тебе извращаться не надо
вот код как надо:

// test class for virtual constructor
class MyClass
{
public:
MyClass()
{
m_nNum = 100;
strcpy(m_szString, "the test string");
}
~MyClass()
{
}
void* operator new(size_t sz, void* pvBlock)
{
return pvBlock;
}
void operator delete (void* pvMem)
{;
}

// make VC++ happy
void operator delete (void* pvMem, void* pvBlock)
{;
}
protected:
int m_nNum;
char m_szString[36];
};

{
char Byte[120];
MyClass* pMyClass = new(Byte) MyClass;
delete pMyClass;
}
