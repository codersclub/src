BOOL GetProcAddresses(HINSTANCE *hLibrary,LPCSTR lpszLibrary,INT nCount,...)
{
va_list va;
va_start( va, nCount );

if ( ( *hLibrary = AfxLoadLibrary( lpszLibrary ) ) != NULL )
{
FARPROC * lpfProcFunction = NULL;
LPSTR lpszFuncName = NULL;
INT nIdxCount = 0;
while ( nIdxCount < nCount )
{
lpfProcFunction = va_arg( va, FARPROC* );
lpszFuncName = va_arg( va, LPSTR );
if ( ( *lpfProcFunction = GetProcAddress( *hLibrary,
lpszFuncName ) ) == NULL )
{
lpfProcFunction = NULL;
return FALSE;
}
nIdxCount++;
}
}
else
{
va_end( va );
return FALSE;
}
va_end( va );
return TRUE;
}

int CPlugThread::Run()
{
HINSTANCE hLib;
typedef void (*PLUGINHANDLER)(DWORD dwCode,HostInfo *pHostInfo,DWORD *pdwResult);
PLUGINHANDLER PluginHandler;

if(!GetProcAddresses(&hLib,m_path,1,&PluginHandler, "PluginHandler"))
{
owner->MessageBox("облом");
AfxEndThread(0,TRUE);
}
DWORD dwResult;
HostInfo HostNfo;

memset(&HostNfo,0,sizeof(HostInfo));
HostNfo.hWnd=owner->m_hWnd;

PluginHandler(1,&HostNfo,&dwResult);

if (hLib!=NULL) FreeLibrary(hLib);

LPTSTR s=(LPTSTR)dwResult;
if (!s) sprintf(s,"Plugin returned value '%d'",dwResult);
owner->SetDlgItemText(IDC_STATUS,s);
delete s;

AfxEndThread(0,TRUE);

return CWinThread::Run();
}
