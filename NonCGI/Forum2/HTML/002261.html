typedef struct tagRGBQUAD
{
char b;
char g;
char r;
char c;
}RQ;

bmp()
{
FILE *B;
char f[50]="file.bmp";
int gdriver = DETECT, gmode;
initgraph(&gdriver, &gmode, "");
B=fopen(f,"r");
fread(&bh,sizeof(bh),1,B);
fread(&bi,sizeof(bi),1,B);
int col=pow(bi.bc,2);
RQ rb[16];
for(int i=0;i<col;i++)
{
fread(&rb[i],sizeof(RQ),1,B);
setrgbpalette(i,rb[i].r,rb[i].g,rb[i].b);
}
fclose(B);

struct lo{

}k;

setviewport(120,70,520,410,1);

fseek(B,bh.offb,SEEK_CUR);

for(y=bi.h;y>=0;y--)
{
for(x=0;x<bi.w-1;x+=2)
