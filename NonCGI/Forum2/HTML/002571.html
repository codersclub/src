Я вот так делал:

LRESULT CPCADWheelDlg::OnSystrayMessage(WPARAM wParam, LPARAM lParam)
{
UINT uMouseMsg = (UINT) lParam;

switch(uMouseMsg)
{
case WM_RBUTTONDOWN:
{
SetForegroundWindow();

CMenuXP *pMenu = new CMenuXP;
pMenu->SetMenuStyle(CMenuXP::STYLE_XP);
pMenu->CreatePopupMenu();
pMenu->SetBackColor(BGCOLOR);
pMenu->SetSelectedBarColor(RGB(182, 189, 210));
pMenu->SetSelectedTextColor(RGB(0, 0, 0));


pMenu->AppendODMenu(0, new CMenuXPText(IDC_SWITCH,
m_bEnabledNow ? "&Включено" : "&Выключено"));
pMenu->AppendODMenu(0, new CMenuXPText(IDC_ABOUT,
"&О программе"));
pMenu->AppendSeparator();

pMenu->AppendODMenu(0, new CMenuXPText(IDCANCEL,
"В&ыход"));

pMenu->CheckMenuItem(IDC_SWITCH, m_bEnabledNow ? MF_CHECKED : MF_UNCHECKED);

POINT point;
GetCursorPos(&point);


pMenu->TrackPopupMenu(TPM_LEFTBUTTON | TPM_RIGHTBUTTON, point.x, point.y, this);

delete pMenu;

break;
}

case WM_LBUTTONDBLCLK:
{
//SetForegroundWindow();
break;
}
case WM_LBUTTONDOWN:
{
//SetForegroundWindow();
break;
}
}
return TRUE;
}

Класс меню - обычный ownerdraw, всем, кстати, рекомендую!
