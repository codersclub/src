Вот, уважаемая, небольшая софтинка - даёт некоторую инфу о удалённом серваке
[code]
#include <stdio.h>
#include <windows.h>
#include <lm.h>
#include <process.h>

int wmain(int argc, wchar_t *argv[])
{
DWORD dwLevel = 102;
LPSERVER_INFO_102 pBuf = NULL;
NET_API_STATUS nStatus;
wchar_t * pszServerName = NULL;

if (argc > 2)
{
fwprintf(stderr, L"Usage: %s [\\\\ServerName]\n", argv[0]);
exit(1);
}
// The server is not the default local computer.
//
if (argc == 2)
pszServerName = argv[1];

//
// Call the NetServerGetInfo function, specifying level 101.
//
nStatus = NetServerGetInfo((LPWSTR)pszServerName,
dwLevel,
(LPBYTE *)&pBuf);
//
// If the call succeeds,
//
if (nStatus == NERR_Success)
{
printf("System: ");
switch (pBuf->sv102_platform_id)
{
case PLATFORM_ID_DOS: printf("DOS\n"); break;
case PLATFORM_ID_OS2: printf("OS/2\n"); break;
case PLATFORM_ID_NT: printf("WinNT\n"); break;
case PLATFORM_ID_OSF: printf("OSF\n"); break;
case PLATFORM_ID_VMS: printf("VMS\n"); break;
default: printf("Unknown system\n"); break;
}

printf("Version: %i.%i\n", pBuf->sv102_version_major, pBuf->sv102_version_minor);

printf("System name: ");
wprintf((wchar_t *)pBuf->sv102_name);
printf("\n");

switch(pBuf->sv102_type)
{
case SV_TYPE_WORKSTATION: printf("A LAN Manager workstation \n"); break;
case SV_TYPE_SERVER: printf("A LAN Manager server \n"); break;
case SV_TYPE_SQLSERVER: printf("Any server running with Microsoft SQL Server \n"); break;
case SV_TYPE_DOMAIN_CTRL: printf("Primary domain controller \n"); break;
case SV_TYPE_DOMAIN_BAKCTRL: printf("Backup domain controller \n"); break;
case SV_TYPE_TIME_SOURCE: printf("Server running the Timesource service \n"); break;
case SV_TYPE_AFP: printf("Apple File Protocol server \n"); break;
case SV_TYPE_NOVELL: printf("Novell server \n"); break;
case SV_TYPE_DOMAIN_MEMBER: printf("LAN Manager 2.x domain member \n"); break;
case SV_TYPE_LOCAL_LIST_ONLY: printf("Servers maintained by the browser \n"); break;
case SV_TYPE_PRINTQ_SERVER: printf("Server sharing print queue \n"); break;
case SV_TYPE_DIALIN_SERVER: printf("Server running dial-in service \n"); break;
case SV_TYPE_XENIX_SERVER: printf("Xenix server \n"); break;
case SV_TYPE_SERVER_MFPN: printf("Microsoft File and Print for NetWare \n"); break;
case SV_TYPE_NT: printf("Windows NT/Windows 2000 workstation or server \n"); break;
case SV_TYPE_WFW: printf("Server running Windows for Workgroups \n"); break;
case SV_TYPE_SERVER_NT: printf("Windows NT/Windows 2000 server that is not a domain controller \n"); break;
case SV_TYPE_POTENTIAL_BROWSER: printf("Server that can run the browser service \n"); break;
case SV_TYPE_BACKUP_BROWSER: printf("Server running a browser service as backup \n"); break;
case SV_TYPE_MASTER_BROWSER: printf("Server running the master browser service \n"); break;
case SV_TYPE_DOMAIN_MASTER: printf("Server running the domain master browser \n"); break;
case SV_TYPE_DOMAIN_ENUM: printf("Primary domain \n"); break;
case SV_TYPE_WINDOWS: printf("Windows 95 or later \n"); break;
case SV_TYPE_ALL: printf("All servers \n"); break;
//case SV_TYPE_TERMINALSERVER: printf("Terminal Server \n"); break;
case SV_TYPE_CLUSTER_NT: printf("Server clusters available in the domain \n"); break;
// case SV_TYPE_CLUSTER_VS_NT: XP: printf("Cluster virtual servers available in the domain \n"); break;
default:
printf("Unknow role\n");
}

printf("Computer's comment: ");
wprintf((wchar_t *)(pBuf->sv102_comment));
printf("\n");

switch(pBuf->sv102_hidden)
{
case SV_VISIBLE: printf("Server is visible. \n"); break;
case SV_HIDDEN: printf("Server is not visible. \n"); break;
}

printf("Announce %i sec\n", pBuf->sv102_announce);
printf("Users who can attempt to log on %i\n", pBuf->sv102_users);
printf("Autodisconnect time %i min\n", pBuf->sv102_disc);

printf("User's path - ");
wprintf((wchar_t *)pBuf->sv102_userpath );
printf("\n");

}
//
// Otherwise, print the system error.
//
else
fprintf(stderr, "A system error has occurred: %d\n", nStatus);
//
// Free the allocated memory.
//
if (pBuf != NULL)
NetApiBufferFree(pBuf);

return 0;
}
