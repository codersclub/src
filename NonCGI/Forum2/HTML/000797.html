Может мне кто-нибудь всё-таки поможет. Чтобы это произошло прилагаю код.

Код на VC++:

hr=m_pConn.CreateInstance(__uuidof(Connection));
hr=m_pConn.CreateInstance("ADODB.Connection");
m_pConn->Open("Data Source=sysCmpnnts;Provider=MSDASQL;User Id=sa;Password=;", "", "", NULL);

//...

char q[100];
_variant_t vRwsAff, value;
_RecordsetPtr pRs;
FieldsPtr pFields;
FieldPtr pField;
int rckey, hdkey;
m_pConn->BeginTrans();

sprintf(q, "exec SP_MNG_CREATE_REC_IN_DETAIL \'%s\', \'%s\', \'%s\', \'%s\', \'%s\'", _bstr_t(name_r), _bstr_t(name_e), _bstr_t(obozn), _bstr_t(cmnt), _bstr_t(src_flag));
pRs=m_pConn->Execute(q, &vRwsAff, adCmdText);
if(vRwsAff.lVal==0)
{
printf("Can't create new record in DETAIL.\n");
m_pConn->RollbackTrans();
return E_ABORT;
}
pFields=pRs->Fields;
pField=pFields->Item[short(0)];
value=pField->Value;
rckey=value.lVal;

if(_bstr_t(src_flag)==_bstr_t("ph"))
{
sprintf(q, "exec SP_MNG_CREATE_REC_IN_PHOTOS %d, %f, %f, %f, %f", rckey, l, w, h, wh);

pRs=m_pConn->Execute(q, &vRwsAff, adCmdText);

//...
}

//...

Хранимые процедуры SP_MNG_CREATE_REC_IN_DETAIL и SP_MNG_CREATE_REC_IN_PHOTOS создают новые записи в таблицах PHOTOS и DETAIL. Стоит обратить внимание, что первый запрос Execute выполняется, а второй нет. В процессе выполнения второго запроса возникает ошибка E_FAIL.

Код на VBasic:

Dim conn As New Connection
conn.Open "sysCmpnnts"
Dim Rs As New Recordset

Set Rs = conn.Execute("exec SP_MNG_CREATE_REC_IN_DETAIL 'oeaiaeia1', 'figovina1', 'fg1-1', 'eiiiaioa?ee', 'ph'", , adCmdText)
If Rs.EOF Then
Return
End If
Dim rckey As Integer
rckey = Rs(0)
Set Rs = conn.Execute("exec SP_MNG_CREATE_REC_IN_PHOTOS " & CStr(rckey) & ", 1.100000, 1.200000, 1.300000, 1.400000", , adCmdText)

Здесь выполняется всё как надо.
