///=======tga.h=============
#ifndef _TGA_H_
#define _TGA_H_
#pragma pack(push,1)

/* TGA-File-Footer Structure */
typedef struct
{
LONG OffsetToExtension;
LONG OffsetToDeveloper;
char Signature[18]; // For "TRUEVISION-XFILE.\0"
} TGAFileFooter;

/* Color map specification */
typedef struct
{
WORD FirstEntryIndex; /* Index of the first color map entry */
WORD Length; /* Total number of color map entries */
BYTE EntrySize; /* Number of bits per entry */
} CMSpec ;

/* Image specification */
typedef struct
{
WORD xOrigin, yOrigin, /* Absolute coordinate for the lower left corner */
Width, Height; /* Width and height in pixels */
BYTE PixelDepth, /* Number of bits per pixel */
Descriptor; /* Image descriptor */
} ISpec ;

/* TGA file header */
typedef struct
{
BYTE IDLength, /* Number of bytes contained in the Image ID Field */
ColorMapType, /* Type of color map (if any) */
ImageType; /* Color-Mapped, True-color or B&W image ;
Uncmopressed or Run-Length encoded */
CMSpec ColorMapSpec; /* Color map specification */
ISpec ImageSpec; /* Image specification */
} TGAFileHeader1;

/* Image Origin : Screen destination first pixel */
#define IO_TOP 0x20
#define IO_BOTTOM 0x00
#define IO_RIGHT 0x10
#define IO_LEFT 0x00
#define IO_MASK 0x30

/* Determine the number of attribute bits per pixel */
#define ACB_MASK 0x0F

/* Color map type */
#define CMT_NOCOLMAP 0 /* No color map included */
#define CMT_COLMAP 1 /* Color map included */

/* Image type */
#define IT_NODATA 0 /* No image data included */
#define IT_UNCPSCOLMAP 1 /* Uncompressed, color mapped image */
#define IT_UNCPSTRUECOL 2 /* Uncompressed, true color image */
#define IT_UNCPSBLKWHIT 3 /* Uncompressed, black & white image */
#define IT_RUNLCOLMAP 9 /* Run-length encoded, color mapped image */
#define IT_RUNLTRUECOL 10 /* Run-length encoded, true color image */
#define IT_RUNLBLKWHIT 11 /* Run-length encoded, black & white image */

typedef struct
{
BYTE IDLength, /* Number of bytes contained in the Image ID Field */
ColorMapType, /* Type of color map (if any) */
ImageType; /* Color-Mapped, True-color or B&W image ; */

WORD FirstEntryIndex; /* Index of the first color map entry */
WORD Length; /* Total number of color map entries */
BYTE EntrySize; /* Number of bits per entry */
WORD xOrigin, yOrigin, /* Absolute coordinate for the lower left corner */
Width, Height; /* Width and height in pixels */
BYTE PixelDepth, /* Number of bits per pixel */
Descriptor; /* Image descriptor */

} TGAFileHeader;

#pragma pack(pop)
#endif // _TGA_H_
///=========end of tga.h====
