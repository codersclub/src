BOOL SetTransparentWindow(HWND hWnd, BYTE Transparent)
{
#ifndef WS_EX_LAYERED
#define WS_EX_LAYERED 0x00080000
#define LWA_COLORKEY 0x00000001
#define LWA_ALPHA 0x00000002
#endif // ndef WS_EX_LAYERED

OSVERSIONINFO os = { sizeof(os) };
GetVersionEx(&os);
BOOL bWin2K = ( VER_PLATFORM_WIN32_NT == os.dwPlatformId && os.dwMajorVersion >= 5 );
if ( !bWin2K ) return FALSE;

typedef BOOL (WINAPI *lpfnSetLayeredWindowAttributes)
(HWND hWnd, COLORREF crKey, BYTE bAlpha, DWORD dwFlags);

static lpfnSetLayeredWindowAttributes pfnSetLayeredWindowAttributes = NULL;
if ( !pfnSetLayeredWindowAttributes )
{

HMODULE hUser32 = GetModuleHandle(_T("USER32.DLL"));
pfnSetLayeredWindowAttributes = (lpfnSetLayeredWindowAttributes)
GetProcAddress(hUser32, "SetLayeredWindowAttributes");


if ( !pfnSetLayeredWindowAttributes ) return FALSE;
}

SetWindowLong(hWnd, GWL_EXSTYLE, GetWindowLong(hWnd, GWL_EXSTYLE) | WS_EX_LAYERED);

pfnSetLayeredWindowAttributes(hWnd, 0, Transparent, LWA_ALPHA);

return TRUE;
}
