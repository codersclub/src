bool decode_header(unsigned long head, MPEGINFO &fr)
{
long framesize, mpeg25, error_protection, bitrate_index, padding, extension, mode_ext;
const long freqs[9] = { 44100, 48000, 32000, 22050, 24000, 16000 , 11025 , 12000 , 8000 };
const int BitRateIndex[2][3][16]=
{
{
{ 0, 32, 64, 96,128,160,192,224,256,288,320,352,384,416,448, 0},
{ 0, 32, 48, 56, 64, 80, 96,112,128,160,192,224,256,320,384, 0},
{ 0, 32, 40, 48, 56, 64, 80, 96,112,128,160,192,224,256,320, 0}
},
{

{ 0, 8, 16, 24, 32, 40, 48, 56, 64, 80, 96,112,128,144,160, 0}
}
};
const float ms_per_frame_array[3][3]={{8.707483f,8.0f,12.0f},{26.12245f,24.0f,36.0f},{26.12245f,24.0f, 36.0f}};

if ((head & 0xffe00000) != 0xffe00000)
return false;
/* if ((head & 0xffff0000) == 0xffff0000)
return false;*/
if(!((head>>17)&3))
return false;
if( ((head>>12)&0xf) == 0xf)
return false;
if( ((head>>10)&0x3) == 0x3 )
return false;

// if( head & (1<<20) )
{
fr.lsf = (head & (1<<19)) ? 0x0 : 0x1;
mpeg25 = 0;
}
/* else
{
fr.lsf = 1;
mpeg25 = 1;
}*/

fr.lay = 4-(head>>17)&3;

if(((head>>10)&0x3) == 0x3)
return false;

if (mpeg25)
fr.sampling_frequency = 6 + ((head>>10)&0x3);
else
fr.sampling_frequency = ((head>>10)&0x3) + (fr.lsf*3);
if (fr.sampling_frequency > 8)
return false;
fr.crc = ((head>>16)&0x1)^0x1;

bitrate_index = ((head>>12)&0xf);
padding = ((head>>9)&0x1);
fr.privat = ((head>>8)&0x1);
fr.mode = ((head>>6)&0x3);
mode_ext = ((head>>4)&0x3);
fr.copyright = ((head>>3)&0x1);
fr.original = ((head>>2)&0x1);
fr.emphasis = head & 0x3;

if(!bitrate_index)
{
return false;
}
fr.bitrate=BitRateIndex[fr.lsf][fr.lay-1][bitrate_index];
switch(fr.lay)
{
case 1:
framesize = (long) fr.bitrate * 12000;
framesize /= freqs[fr.sampling_frequency];
framesize = ((framesize+padding)<<2)-4;
break;
case 2:
framesize = (long) fr.bitrate * 144000;
framesize /= freqs[fr.sampling_frequency];
framesize += padding - 4;
break;
case 3:
framesize = (long) fr.bitrate * 144000;
framesize /= freqs[fr.sampling_frequency]<<(fr.lsf);
framesize = framesize + padding - 4;
break;
default:
return false;
}
fr.framecount = fr.size/(framesize +4-padding);
fr.ms = (int)fr.size/(framesize +5-padding)*ms_per_frame_array[1][fr.sampling_frequency];
fr.sampling_frequency=freqs[fr.sampling_frequency];
return 1;
}
