Ну так и создавай док и вьюв ручками в чем проблема то?
вот код (схематично канешна)

CDocument* CreateDoc(CRuntimeClass* pDocClass)
{
if (pDocClass == NULL)
{
return NULL;
}
CDocument* pDocument = (CDocument*)pDocClass->CreateObject();
if (pDocument == NULL)
{
TRACE1("Warning: Dynamic create of document type %hs failed.\n",
m_pDocClass->m_lpszClassName);
return NULL;
}
return pDocument;
}

//create view
CreateView(CDocument* pDoc, CRuntimeClass* pViewClass);
{
CCreateContext context;
context.m_pCurrentFrame = NULL;
context.m_pCurrentDoc = pDoc;
context.m_pNewViewClass = pViewClass;
context.m_pNewDocTemplate = NULL;

CWnd* pView = (CWnd*)context.m_pNewViewClass->CreateObject();

if (pView == NULL)
{
TRACE1("Warning: Dynamic create of view type %hs failed.\n",
pContext->m_pNewViewClass->m_lpszClassName);
return NULL;
}
ASSERT_KINDOF(CWnd, pView);

// views are always created with a border!
if (!pView->Create(NULL, NULL, AFX_WS_DEFAULT_VIEW,
CRect(0,0,0,0), this, nID, &context))
{
TRACE0("Warning: could not create view for frame.\n");
return NULL; // can't continue without a view
}
return pView;
}

Ну а дальше используй его как обычное дочернее окно. Только не забудь в документе сказать m_bAutoDelete = TRUE да и вьюв придеться чуток переписать - потому как он зарраза считает что всегда создается на фрэйме и юзает фреймовские функции иногда - ну это тебе отладчик скажет где.
