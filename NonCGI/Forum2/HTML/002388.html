#include<windows.h>
#include<iostream.h>

PROCESS_INFORMATION pi;

bool StartSlaveProcess(char *name,HANDLE hReadPipe,HANDLE hWritePipe)
{
STARTUPINFO si;
char buf[512];
bool flag;

GetStartupInfo(&si);

strcpy(buf,name);

ZeroMemory( &si, sizeof(si) ); // fill memory with zero.
si.cb = sizeof(si);
si.wShowWindow = SW_NORMAL;
si.dwFlags = STARTF_USESHOWWINDOW;
si.hStdOutput=hWritePipe;
si.hStdInput=hReadPipe;

flag=CreateProcess(
NULL,
buf,
NULL, // security attributes
NULL, // security attributes
TRUE, //Inherited handles have the same value and access privileges as the original handles.
NORMAL_PRIORITY_CLASS, // CreationFlags
NULL,
NULL,
&si,
&pi);

return flag;
}

Ты забыл вставить следующий код:

if(!DuplicateHandle(hProc,hParentStdin,hProc,&InPipe,
0,FALSE,DUPLICATE_SAME_ACCESS))
return false;

CloseHandle(hParentStdin);

if(!DuplicateHandle(hProc,hParentStdout,hProc,&OutPipe,
0,FALSE,DUPLICATE_SAME_ACCESS))
return false;

CloseHandle(hParentStdout);

То есть, ты должен продублировать хэндлы тех концов канала, которые передаешь в другой процесс, а также сказать ему, что он должен наследовать хэндлы, переданные родительским процессом.
