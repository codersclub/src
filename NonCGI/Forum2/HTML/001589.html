Пришлось поковырятся в MSDN и немного выпрямить рукию В конце получилось вот это:

======================================================================

#include <stdio.h>
#include <memory.h>

#define LD_SIZE 10

// Convert 10-byte long double buffer to standard double (8-byte).
double ld2double(unsigned char *ldbuf)
{
long double ld;
unsigned char b[LD_SIZE];

memcpy(b, ldbuf, LD_SIZE);

// This moves the contents of the buffer into the floating point
// register, which then then takes care of the automatic convertion
// back to a 8-byte long double
_asm {
fld TBYTE PTR b;
fstp ld;
}

return (double)ld;
}

void main(void)
{
unsigned char buf[10] = {0x0A, 0xD7, 0xA3, 0x70, 0x3D, 0x0A, 0xD7, 0xA3, 0xF8, 0x3F};
double d;

d = ld2double(buf);

printf("10-byte Long Double to standard Double (8-byte) converting example.\n\n");
printf("Input: 0.01 (long double buffer)\n");
printf("Result: %.2f (double)\n\n", d);
}

======================================================================

Только к сожалению приходится делать промежуточный буффер в самой функции потому, что в VC++ встроенный ассемблер не хочет дружить с пойнтером LD буффера. Может есть вариант поэлегантнее, чтобы обойтись без промежуточного буффера?
