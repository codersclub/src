uses ShellAPI;

function GetMyDocsPath: string;
var
Buf: array[0..MAX_PATH];
PIDL: PItemIDList;
begin
SHGetSpecialFolderLocation(0, CSIDL_PERSONAL, PIDL);
SHGetPathFromIDList(PIDL,@Buf[0]);
Result:=PChar(@Buf[0]);
end;
