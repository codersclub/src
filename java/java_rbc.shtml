import java.io.*;
import java.util.*;
import java.net.*;
import java.text.*;
import java.sql.*;

public class ConnectManager {

  public static Connection getConnectionToRBC() throws IOException {
       Connection con = null;
       Locale.setDefault(new Locale("ru","RU"));
       try {
	Class.forName("sun.jdbc.odbc.JdbcOdbcDriver");
	con = DriverManager.getConnection("jdbc:odbc:rbc", "", "");
       } catch(Exception ex) {
       }
       return con;
  }

  public static void closeConnection(Connection con) throws IOException {
       try {
	con.close();
       } catch (SQLException e) {
       }
  }

  public static String toUnicode(String str) throws IOException {
       try {
	String charset="windows-1251";
	byte[] buf = new byte[str.length()];
	str.getBytes(0, str.length(), buf, 0);
	return new String(buf);
       } catch(Exception e) {
	return "Error toUnicode";
       }
  }
}
