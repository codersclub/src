   Private Sub Form_Load()

      Dim ShadowX
      Dim ShadowY

                                   'Очищаем Form1
      Form1.ScaleMode = 3          'Устанавливаем Scalemode в pixel
      Form1.ForeColor = vb3DShadow 'Меняем основной цвет Form1
                                   'на цвет тени (тёмно серый)
      ShadowY = 5                  'Устанавливаем высоту тени
      ShadowX = 5                  'Устанавливаем ширину тени
      For I = 0 To 5
         CurrentX = ShadowX + I
         CurrentY = ShadowY + I
         If I = 5 Then Form1.ForeColor = vbWhite 'Меняем цвет на Белый
         Form1.Print "Text in 3D. No VBX/OCX/API!"
      Next
   End Sub
